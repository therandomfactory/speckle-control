13 ## Documented proc \c geminiConnect .
14 # \param[in] scope Telescope id , north or south
15 #
16 # Connect to the telemery server
17 #
18 # Globals :
19 # GEMINICFG - Gemini specific proprties
20 #
21 proc geminiConnect { scope } {
22 global GEMINICFG
23  set handle -1
24  set handle [socket $GEMINICFG($scope,ip) $GEMINICFG($scope,port)]
25  fconfigure $handle -buffering line -blocking 0
26  if { $handle < 0 } {
27  errordialog "Failed to connect to Gemini service at $GEMINICFG($scope,ip) port $GEMINICFG($scope,port) "
28  } else {
29  debuglog "Connected to Gemini $GEMINICFG($scope,ip) port $GEMINICFG($scope,port) - OK"
30  set GEMINICFG(handle) $handle
31  }
32  puts $GEMINICFG(handle) "get airmass"
33  gets $GEMINICFG(handle) rec
34  return $handle
35 }

53 ## Documented proc \c updateGeminiTelemetry .
54 #
55 # Update telemetry by querying server
56 #
57 # Globals :\n
58 # GEMINICFG - Gemini specific proprties\n
59 # GEMINI - Gemini specific items\n
60 # TELEMETRY - Global telemetry array for headers and database
61 #
62 proc updateGeminiTelemetry { } {
63 global GEMINI GEMINICFG TELEMETRY SCOPE
64  set all [lsort [array names GEMINI]]
65  foreach t $all {
66  puts $GEMINICFG(handle) "get $t\n"
67  }
68  after 500
69  while { [gets $GEMINICFG(handle) rec] > -1 } {
70  if { [info exists GEMINI([lindex $rec 1])] } {
71  set TELEMETRY($GEMINI([lindex $rec 1])) [lindex $rec 2]
72  debuglog "Got $rec"
73  } else {
74  debuglog "Got unknown $rec"
75  }
76  }
77  echoGeminiTelemetry
78  set SCOPE(ProgID) $TELEMETRY($GEMINI(progamid))
79  set SCOPE(ra) $TELEMETRY($GEMINI(targetra))
80  set SCOPE(dec) $TELEMETRY($GEMINI(targetdec))
81  set SCOPE(target) $TELEMETRY($GEMINI(targetname))
