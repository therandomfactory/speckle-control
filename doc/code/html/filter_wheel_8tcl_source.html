123  }

130 #

133 #
134 proc findWheels { } {
135 global FWSERIAL
136  set fw [split [exec lsusb] \n]
137  set id 1

139  if { [lindex $i 6] == "Newport" } {

218 ## Documented proc \c setOrielFilter .
219 # \param[in] id Wheel identifier
220 # \param[in] n Filter position
221 #
222 # Call low level code to move filter to selected position
223 #
224 proc setOrielFilter { id n } {
225  oriel_write_cmd $id FILT?
226  after 200
227  set res [oriel_read_result $id]
228  set cpos [string range $res 4 4]
229  set npos [expr $n - $cpos]
230  while { $npos > 0 } {
231  oriel_write_cmd $id NEXT
232  incr npos -1
233  after 2000
234  set res [oriel_read_result $id]
235  }
236  while { $npos < 0 } {
237  oriel_write_cmd $id PREV
238  incr npos 1
239  after 2000
240  set res [oriel_read_result $id]
241  }
242  oriel_write_cmd $id FILT?
243  after 200
244  set res [oriel_read_result $id]
245  set cpos [string range $res 4 4]
246  if { $cpos == $n } {
247  return $n
248  }
249  return -1
250 }
