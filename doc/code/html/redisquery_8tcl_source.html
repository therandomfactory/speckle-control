20 ## Documented proc \c redisReader .
21 # \param[in] fh Socket handle of Redis server connection
22 #
23 # Read any new data from the Redis server
24 #
25 # Globals :
26 # TELEMETRY - Array of telemetry items for headers and database usage
27 #
28 proc redisReader { fh } {
29 global TELEMETRY
30  while { [gets $fh res] > -1} {
31 ### debuglog "redis : $res"
32  if { [lindex $res 0] == "timestamp" || [string range $res 0 3] == "tcs." || [string range $res 0 3] == "dcs." || [string range $res 0 3] == "oss."} {
33  set id $res
34  gets $fh res ; gets $fh res
35  set TELEMETRY($id) [string trim $res "\{\}"]
36  }
37  }
38 }
