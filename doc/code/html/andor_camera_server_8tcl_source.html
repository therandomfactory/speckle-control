23 ## Documented proc \c cAndorSetProperty .
24 # \param[in] arm Name of intrument arm red/blue
25 # \param[in] prop Camera property name
26 # \param[in] val Value to set the the property to
27 # \param[in] val2 Optional second value for HSSpeed usage
28 #
29 # This wraps the low level (C) shared library calls which set camera properties
30 # e.g it will call methods like SetExpsoureTime. It records the results of
31 # sucessfull calls in the ANDOR_CFG array for easy access from the tcl side
32 #
33 #
34 # Globals :\n
35 # ANDOR_CFG - Andor camera properties
36 # ANDOR_ARM - Instrument arm this camera is installed in red/blue
37 #
38 proc cAndorSetProperty { cam prop val {val2 ""} } {
39 global ANDOR_CFG ANDOR_ARM
40  if { $prop == "HSSpeed" } {
41  if { $val == 0 } { set res [andorSetProperty $cam HSSpeed 0 $val2] ; set prop EMHSSpeed}
42  if { $val == 1 } { set res [andorSetProperty $cam HSSpeed 1 $val2]}
43  } else {
44  set res [andorSetProperty $cam $prop $val]
45  }
46  if { $res == "" } {
47  set ANDOR_CFG($cam,$prop) $val
48  set ANDOR_CFG($ANDOR_ARM,$prop) $val
49  } else {
50  set ANDOR_CFG($cam,$prop) "fail -$res"
51  set ANDOR_CFG($ANDOR_ARM,$prop) "fail - $val"
52  }
53  return $res
54 }

325  appendHeader $SPECKLE_DATADIR/[set ANDOR_CFG(imagename)]_red.fits

664 # ANDOR_EMCCD - Number of EMCCD amplifier\n
665 # ANDOR_CODE - Indexes to arrays of speed identifiers

667 proc configReadout { amp hsspeed vsspeed vsamplitude preampgain emgain emgainmode } {

827  set ANDOR_ROI(ys) $ys

892  setframe { configureFrame [lindex $msg 1] ; puts $sock "OK"}
893  setbinning { set ANDOR_CFG($CAM,hbin) [lindex $msg 1] ; set ANDOR_CFG($CAM,vbin) [lindex $msg 2] ; set ANDOR_CFG(binning) [lindex $msg 1] ;puts $sock "OK"}
894  scalepeak { set ANDOR_CFG(scalepeak) [lindex $msg 1] ; puts $sock "OK"}
895  fitsbits { set ANDOR_CFG(fitsbits) [lindex $msg 1] ; puts $sock "OK"}
896  whicharm { puts $sock $ANDOR_ARM}
897  forceroi { forceROI [lindex $msg 1] [lindex $msg 2] [lindex $msg 3] [lindex $msg 4] ; puts $sock "OK"}
898  locatestar { puts $sock "[locateStar [lindex $msg 1] [lindex $msg 2]]"}
899  datadir { set SPECKLE_DATADIR [lindex $msg 1] ; puts $sock "OK"}
