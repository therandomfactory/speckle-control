113 # PICOS - Array of pico configuration data
114 #
115 proc picoCommand { axis cmd } {
116 global PICOS
117  debuglog "Commanding $axis picomotor - $cmd"
118  if { $PICOS(sim)} {
119  set rec "SIM $axis $cmd"
120  set ctype [lindex $cmd 0]
121  if { $ctype == "POS"} {
122  set PICOS($axis,position) [lindex [split $cmd "= "] 2]
123  }
124  if { $ctype == "REL"} {
125  set delta [lindex [split $cmd "= "] 2]
126  set PICOS($axis,position) [expr $PICOS($axis,position) + $delta]
127  }
128  set PICOS($axis,current) $PICOS($axis,position)
129  } else {
130  debuglog "Sending PICO command $axis $cmd"

152 proc picoSet { axis par {value ""} } {
153 global PICOS
154  debuglog "PICO command set $axis $par $value"
155  switch $par {
156  disable { set res [picoCommand $axis AB] }
157  stop { set res [picoCommand $axis ST] }
158  position { set res [picoCommand $axis "PA $value"] }

160  offset { set res [picoCommand $axis "PR $value"] }
161  velocity { set res [picoCommand $axis "VA $value] }
162  reset { set res [picoCommand $axis "RS"] }
163  in { set res [picoCommand $axis "PA $PICOS($axis,in)"] }
164  out { set res [picoCommand $axis "PA $PICOS($axis,out)"] }
165  home { set res [picoCommand $axis "PA $PICOS($axis,home)"] }
166  }
